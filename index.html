def json_dumps(*args, **kwargs):
    encode_binary = kwargs.get('encode_binary', False)
    if 'encode_binary' in kwargs:
        del kwargs['encode_binary']
    return json.dumps(*args, default=default(encode_binary), **kwargs)

def default(encode_binary=False):
    def _default(obj):
        if isinstance(obj, ObjectId):
            return str(obj)
        if isinstance(obj, DBRef):
            return str(obj.id)
        if isinstance(obj, datetime):
            if obj.utcoffset() is not None:
                obj = obj - obj.utcoffset()
            return obj.isoformat()
        if hasattr(obj, '__json__'):
            return obj.__json__(encode_binary)
        raise TypeError("%r is not JSON serializable" % obj)
    return _default
